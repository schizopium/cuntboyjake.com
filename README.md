<script data-scope="dob-mask" id="i0qpmn">
  (function initDobMask() {
    const dob = document.getElementById('dob');
    if (!dob) return;

    function formatDigitsToDob(digits) {
      if (!digits) return '';
      const mm = digits.slice(0, 2);
      const dd = digits.length > 2 ? digits.slice(2, 4) : '';
      const yyyy = digits.length > 4 ? digits.slice(4, 8) : '';
      let out = mm;
      if (digits.length > 2) out += '/' + dd;
      if (digits.length > 4) out += '/' + yyyy;
      return out;
    }

    function caretPosForDigitIndex(masked, idx) {
      let count = 0;
      for (let i = 0; i < masked.length; i++) {
        if (/\d/.test(masked[i])) {
          count++;
          if (count === idx) return i + 1;
        }
      }
      return masked.length;
    }

    function applyMaskFromRaw(raw, selStart) {
      const digitsBeforeCaret = (raw.slice(0, selStart).match(/\d/g) || []).length;
      const digits = (raw.match(/\d/g) || []).join('').slice(0, 8);
      const masked = formatDigitsToDob(digits);
      dob.value = masked;
      const newPos = caretPosForDigitIndex(masked, digitsBeforeCaret);
      try {
        dob.setSelectionRange(newPos, newPos);
      } catch (_) {}
    }

    function insertDigitsAtSelection(insertDigits) {
      const currentDigits = (dob.value.match(/\d/g) || []).join('');
      const startDigitsIdx = (dob.value.slice(0, dob.selectionStart).match(/\d/g) || []).length;
      const endDigitsIdx = (dob.value.slice(0, dob.selectionEnd).match(/\d/g) || []).length;
      const newDigits = (currentDigits.slice(0, startDigitsIdx) + insertDigits + currentDigits.slice(endDigitsIdx)).slice(0, 8);
      const masked = formatDigitsToDob(newDigits);
      dob.value = masked;
      const newPos = caretPosForDigitIndex(masked, startDigitsIdx + insertDigits.length);
      try {
        dob.setSelectionRange(newPos, newPos);
      } catch (_) {}
    }

    dob.addEventListener('beforeinput', (e) => {
      const deletion = /^delete/.test(e.inputType);
      if (deletion || e.inputType === 'insertCompositionText') return;

      if (e.inputType === 'insertText') {
        const ch = e.data || '';
        if (!/^\d$/.test(ch)) {
          e.preventDefault();
          return;
        }
        const currentDigitsCount = (dob.value.match(/\d/g) || []).length;
        const selLenDigits = (dob.value.slice(dob.selectionStart, dob.selectionEnd).match(/\d/g) || []).length;
        if (currentDigitsCount - selLenDigits >= 8) {
          e.preventDefault();
          return;
        }
      }

      if (e.inputType === 'insertFromPaste') {
        e.preventDefault();
      }
    });

    dob.addEventListener(
      'input',
      (e) => {
        e.stopPropagation();
        const raw = dob.value;
        const selStart = dob.selectionStart || raw.length;
        applyMaskFromRaw(raw, selStart);
      },
      true
    );

    dob.addEventListener('paste', (e) => {
      e.preventDefault();
      const text = (e.clipboardData || window.clipboardData).getData('text') || '';
      const digits = (text.match(/\d/g) || []).join('').slice(0, 8);
      insertDigitsAtSelection(digits);
    });

    dob.addEventListener('keydown', (e) => {
      const allowedKeys = [
        'Backspace',
        'Delete',
        'ArrowLeft',
        'ArrowRight',
        'Home',
        'End',
        'Tab'
      ];
      if (allowedKeys.includes(e.key)) return;
      if (!/^\d$/.test(e.key)) {
        e.preventDefault();
      } else {
        const currentDigitsCount = (dob.value.match(/\d/g) || []).length;
        const selLenDigits = (dob.value.slice(dob.selectionStart, dob.selectionEnd).match(/\d/g) || []).length;
        if (currentDigitsCount - selLenDigits >= 8) {
          e.preventDefault();
        }
      }
    });

    applyMaskFromRaw(dob.value || '', (dob.value || '').length);
  })();
